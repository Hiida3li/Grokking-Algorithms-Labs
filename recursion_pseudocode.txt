def fact(x):
    # Trace when function is called
    print(f"ENTERING FACT({x})")
    if x == 1:
        print(f"BASE CASE IS REACHED: returning 1")
        return 1
    else:
        result = x * fact(x-1)
        print(f"Returning {result} for fact({x})")
    return result
